package com.example.myapplication;

import androidx.appcompat.app.AppCompatActivity;


import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.content.Intent;
import android.widget.TextView;

import com.example.createeventpage.R;

public class createeventpage extends AppCompatActivity
{
    String eventName, eventDate, eventTime, eventLocation, description, gePoint;
    TextView eventNameText, eventDateText, eventTimeText, eventLocationText, descriptionText, gePointText;
    private Button button;
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.createeventpage);

        button = (Button) findViewById(R.id.button);
        eventNameText = (TextView) findViewById(R.id.eventNameText);
        eventDateText = findViewById(R.id.eventDateText);
        eventLocationText = findViewById(R.id.eventLocationText);
        eventTimeText = findViewById(R.id.eventTimeText);
        descriptionText = findViewById(R.id.descriptionText);
        gePointText = findViewById(R.id.gePointText);

        button.setOnClickListener(new View.OnClickListener()
        {
            @Override
            public void onClick(View v)
            {
                openActivity2();
            }
        });
    }

    public void createeventClicked(View view)
    {
        eventName = eventNameText.getText().toString();
        eventDate = eventDateText.getText().toString();
        eventTime = eventTimeText.getText().toString();
        eventLocation = eventLocationText.getText().toString();
        description = descriptionText.getText().toString();
        gePoint = gePointText.getText().toString();
    }

    public void openActivity2()
    {
        Intent intent = new Intent(this, createeventpage.class);
        startActivity(intent);

    }
}
